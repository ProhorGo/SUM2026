#include <stdio.h>
#include <windows.h>
#include <math.h>

#define MAX 5
INT P[MAX] = {1, 2, 3, 4, 5};
BOOL IsParity = TRUE;

VOID Swap( INT *a, INT *b )
{
  INT tmp = *a;
  *a = *b;
  *b = tmp;
}

VOID Parity( VOID )
{
  INT i, j;

  for (i = 0; i < MAX; i++)
    for (j = i + 1; j < MAX; j++)
      if (P[i] > P[j])
        IsParity = !IsParity;
}

VOID Store( VOID )
{
  FILE *F;
  INT i;

  F = fopen("qqq.txt", "a");
  if (F == NULL)
    return;
  Parity();
  for (i = 0; i < MAX - 1; i++)
    fprintf(F, "%d, ", P[i]);
  fprintf(F, "%d - parity: %s\n", P[MAX - 1], IsParity ? "even" : "odd");
  fclose(F);
}                                     
  
VOID Go( INT Pos )
{
  INT i;

  if (Pos == MAX)
    Store();
  else
  {
    for (i = Pos; i < MAX; i++)
    {
      Swap(&P[Pos], &P[i]);
      Go(Pos + 1);
      Swap(&P[Pos], &P[i]);
    }
  }
}

void main( void )
{
  Go(0);
}
