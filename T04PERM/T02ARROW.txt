#include <stdio.h>
#include <time.h>
#include <math.h>
#include <windows.h>
#include <winnls.h>
#include <stdlib.h>
#define WND_CLASS_NAME "cgsg209"
#define PI 3.1415926535897931

LRESULT CALLBACK MyWindowFunc( HWND hWnd, UINT Msg,
                              WPARAM wParam, LPARAM lParam);

VOID DrawArrow( HDC hDC, INT cx, INT cy, FLOAT L, FLOAT W, FLOAT angle, INT r, INT g, INT b );

VOID ArrowMas( HDC hDC, INT W, INT H, INT step, INT Long, INT Weight, POINT pt );

INT WINAPI WinMain( HINSTANCE hInstance, HINSTANCE hPrevInstanse, 
                   CHAR *CmdLine, INT ShowCmd )
{
  HWND hWnd;
  WNDCLASS wc;
  MSG msg;

  wc.style = 0;
  wc.cbClsExtra = 0;
  wc.cbWndExtra = 0;
  wc.hbrBackground = (HBRUSH)COLOR_WINDOW;
  wc.hCursor = LoadCursor(NULL, IDC_HAND);
  wc.hIcon = LoadIcon(NULL, IDI_SHIELD);
  wc.hInstance = hInstance;
  wc.lpszMenuName = NULL;
  wc.lpszClassName = WND_CLASS_NAME;
  wc.lpfnWndProc = MyWindowFunc;

  if (!RegisterClass(&wc))
  {
    MessageBox(NULL, "error", "ERROR", MB_ICONERROR | MB_OK);
    return 0;
  }

  hWnd = CreateWindow(WND_CLASS_NAME, "SummerPractice2026", WS_CLIPCHILDREN | WS_OVERLAPPEDWINDOW | WS_VISIBLE,
    100, 100, 500, 300, NULL, NULL, hInstance, NULL);
 
  CreateWindow("BUTTON", "CLOSE", WS_CHILD | WS_VISIBLE,
    10, 10, 100, 30, hWnd, (HMENU)123, hInstance, NULL);

  while (GetMessage(&msg, NULL, 0, 0))
  {
    TranslateMessage(&msg);       
    DispatchMessage(&msg);
  }
  return msg.wParam;
}

VOID DrawArrow( HDC hDC, INT cx, INT cy, FLOAT L, FLOAT Wh, FLOAT angle, INT r, INT g, INT b )
{
  POINT pnts[4];
  HPEN hPen;
  HBRUSH hBr;

  pnts[0].x = cx + (int)(L * cos(angle));
  pnts[0].y = cy + (int)(L * sin(angle));
  pnts[1].x = cx - (int)(Wh * sin(angle));
  pnts[1].y = cy - (int)(Wh * cos(angle));
  pnts[2].x = cx + (int)(Wh * sin(angle));
  pnts[2].y = cy + (int)(Wh * cos(angle));
 
  hPen = CreatePen(PS_SOLID, 5, RGB(r, g, b));
  hBr = CreateSolidBrush(RGB(r, g, b));
  SelectObject(hDC, hPen);
  SelectObject(hDC, hBr);
  Polygon(hDC, pnts, 3);
  DeleteObject(hBr);
  DeleteObject(hPen);
} 

VOID ArrowMas( HDC hDC, INT W, INT H, INT step, INT Long, INT Weight, POINT pt )
{
  INT i, j;

  for (i = 0; i < H; i += step)
    for (j = 0; j < W; j += step)
      DrawArrow(hDC, j, i, Long, Weight, atan2(pt.y - i, pt.x - j),
        rand() % 255, rand() % 255, rand() % 255); 
}
                                             
LRESULT CALLBACK MyWindowFunc( HWND hWnd, UINT Message,
                              WPARAM wParam, LPARAM lParam)
{
  HDC hDC;
  PAINTSTRUCT ps;
  static INT W, H;
  static HDC hMemDC;
  static HBITMAP hBm;
  POINT pt;
  
  switch (Message)
  {
  case WM_CREATE:
    hDC = GetDC(hWnd);
    hMemDC = CreateCompatibleDC(hDC);
    ReleaseDC(hWnd, hDC);
    SetTimer(hWnd, 30, 100, NULL);
    return 0;
  case WM_COMMAND:
    SendMessage(hWnd, WM_CLOSE, 0, 0);
    return 0;
  case WM_SIZE:         
    H = HIWORD(lParam);
    W = LOWORD(lParam);
    if (hBm != NULL)
      DeleteObject(hBm);
    hDC = GetDC(hWnd);
    hBm = CreateCompatibleBitmap(hDC, W, H);
    ReleaseDC(hWnd, hDC);
    SelectObject(hMemDC, hBm);
    InvalidateRect(hWnd, NULL, FALSE);                           
    return 0;
  case WM_TIMER:
    InvalidateRect(hWnd, NULL, FALSE);
    return 0;
  case WM_LBUTTONDOWN:
    InvalidateRect(hWnd, NULL, FALSE);
    return 0; 
  case WM_PAINT:
    hDC = BeginPaint(hWnd, &ps);

    SelectObject(hMemDC, GetStockObject(WHITE_PEN)); 
    SelectObject(hMemDC, GetStockObject(WHITE_BRUSH));
    Rectangle(hMemDC, 0, 0, W, H);

    GetCursorPos(&pt);
    ScreenToClient(hWnd, &pt);
    ArrowMas(hMemDC, W, H, 20, 20, 5, pt);
                               
    BitBlt(hDC, 0, 0, W, H, hMemDC, 0, 0, SRCCOPY);
    EndPaint(hWnd, &ps);
    return 0;
  case WM_ERASEBKGND:
    return 1;
  case WM_DESTROY:
    DeleteObject(hBm);
    DeleteDC(hMemDC);
    KillTimer(hWnd, 30);
    PostQuitMessage(0);
    return 0;
  }
  return DefWindowProc(hWnd, Message, wParam, lParam);
}
