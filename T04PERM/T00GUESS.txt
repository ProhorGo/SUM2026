#include <stdio.h>
#include <windows.h>
#include <string.h>

void main( void )
{
  int a = 0, b = 100, m = (a + b) / 2, res;
  char str[100];

  if (MessageBox(NULL, "Zagaday number", "IGRA", MB_OKCANCEL) == IDOK)
  {
    while (a < b)
    {
      m = (a + b) / 2;
      sprintf(str, "Number bolshe %i", m);
      res = MessageBox(NULL, str, "IGRA", MB_YESNOCANCEL);
      if (res == IDYES)
        a = m + 1;
      else if (res == IDNO)
        b = m;
      else if (res == IDCANCEL)
        break;
    }
  }
  sprintf(str, "Number is %i", m);
  if (MessageBox(NULL, str, "IGRA", MB_OKCANCEL) == IDOK)
    MessageBox(NULL, "URA!!!", "IGRA", MB_OK);
  else
    MessageBox(NULL, "ERROR!!!", "IGRA", MB_OK);

}
