#include <stdio.h>
#include <windows.h>
typedef DOUBLE DBL;
#define MAX 10

INT P[MAX];
BOOL IsParity;
DBL A[MAX][MAX];
INT N;
DBL Det;

VOID Swap( INT *a, INT *b )
{
  INT tmp = *a;
  *a = *b;
  *b = tmp;
}

BOOL LoadMatrix( CHAR *FileName )
{
  FILE *F;
  INT i, j;
 
  N = 0;
  if ((F = fopen(FileName, "r")) == NULL)
    return FALSE;
  fscanf(F, "%d", &N);
  if (N > MAX)
    N = MAX;
  else
    if (N < 0)
      N = 0;
  for (i = 0; i < N; i++)
    for (j = 0; j < N; j++)
      fscanf(F, "%lf", &A[i][j]);
  fclose(F);
  return TRUE;
}
                  
INT Parity( VOID )
{
  INT i, j;
  IsParity = TRUE;

  for (i = 0; i < N; i++)
    for (j = i + 1; j < N; j++)
      if (P[i] > P[j])
        IsParity = !IsParity;
  if (IsParity)
    return 1;
  return -1;
}  

VOID Store( VOID )
{ 
  INT i;
  DBL a = 1;

  for (i = 0; i < N; i++)
    a *= A[i][P[i]];
    
  Det += Parity() * a;
}   

VOID Go( INT Pos )
{
  INT i;

  if (Pos == N)
    Store();
  else
  {
    for (i = Pos; i < N; i++)
    {
      Swap(&P[Pos], &P[i]);
      Go(Pos + 1);
      Swap(&P[Pos], &P[i]);
    }
  }
}

VOID main( VOID )
{
  CHAR *FileName = "IN.TXT";
  INT i;

  LoadMatrix(FileName);
  for (i = 0; i < N; i++)
    P[i] = i;
  Det = 0;
  Go(0);  
  printf("Solution: %f", Det);
  _getch();
}
